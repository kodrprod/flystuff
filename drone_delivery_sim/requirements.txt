# ===========================================================================
# Python dependencies for the drone delivery simulation.
#
# Every package below installs as a PREBUILT WHEEL (no compiler needed) on
# macOS (Apple Silicon AND Intel), Windows and Linux, for Python 3.10-3.13.
# Just run:   pip install -r requirements.txt
#
# IMPORTANT: this uses opencv-contrib-python (NOT plain opencv-python). The
# "contrib" build is the one that ships cv2.aruco, which this project needs.
# Do NOT also install opencv-python -- having both installed causes conflicts.
#
# Why opencv is pinned to 4.10.0.84 (and not something newer): its macOS wheel
# supports older systems (back to macOS 11 on Apple Silicon), while opencv
# 4.11+ wheels require macOS 13+ and are missing for some Python versions. On
# an older Mac those newer pins fail to install and you get the dreaded
# "ModuleNotFoundError: No module named 'cv2'". 4.10.0.84 just works.
# ===========================================================================

numpy==2.2.6
opencv-contrib-python==4.10.0.84
matplotlib==3.10.9
imageio==2.37.3
imageio-ffmpeg==0.6.0

# --- 3D world engine (OPTIONAL -- intentionally NOT installed by default) ---
# PyBullet gives nicer camera feeds and exact collision, BUT it publishes NO
# macOS wheels (only Linux), so on a Mac "pip install pybullet" has to COMPILE
# it from source -- which is slow, needs Xcode build tools, and often fails.
#
# The simulation runs perfectly WITHOUT it: it automatically falls back to a
# built-in pure-numpy 3D engine. That is why pybullet is left out of the
# install above, so the core setup can never be blocked by it.
#
# If you are on Linux, or have a working compiler on macOS, and want the
# PyBullet backend, install it separately:
#
#     pip install pybullet==3.2.7
# ---------------------------------------------------------------------------
